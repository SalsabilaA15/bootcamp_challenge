##Konfigurasi

ada di config.php
``
global $config;
$config = array();
define("BASE","http://localhost/bootcamp");
define("BASEDIR", __DIR__ . '/');
define("PROJECTPATH", dirname(__DIR__) . '/bootcamp');
$config['dbname']= 'bootcamp';
$config['host']= 'localhost';
$config['password']= '';
$config['dbuser']= 'root';
``
